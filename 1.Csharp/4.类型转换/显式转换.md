# 讲解
```C#
namespace 显式转换
{
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("类型转换，显式转换");
            
            //显式转换→手动处理，强制转换
            
            #region 括号强转
            //作用：一般情况下，将高精度的类型强制转换为低精度
            //语法：变量类型 变量名 = （变量类型）变量；
            //注意：精度问题、范围问题
            
            //相同大类的整形
            //有符号的整形
            sbyte sb = 1;
            short s = 2;
            int i = 3;
            long l = 4;
            
            ///s = i;(报错，因为不能隐式转换，小范围装大范围）
            
            ///此处int类型强制转换为short类型
            s = (short)i;
            Console.WriteLine(s);
            
            i = (int)l;
            sb = (sbyte)s;
            
            ///括号强转可能会出现范围问题造成的异常
            int i2 = 40000;
            s = (short)i2;
            Console.WriteLine(s);
            
            //无符号整形
            byte b = 1;
            uint ui = 1;
            b = (byte)ui;
            
            //浮点之间
            float f0 = 1.1f;
            double d = 1.11;
            
            f0 = (float)d;
            ///此类型超出范围影响不大，相当于只是精度丢失
            
            //无符号和有符号之间
            uint ui2 = 1;
            int i3 = -1;
            ui2 = (uint)i3;
            Console.WriteLine(i3);
            ///二进制造成的错误，因为int前面一位是符号位
            ///在强转的时候一定要注意范围，不然得到的结果有异常
            
            //浮点和整数
            i2 = (int)1.24f;
            Console.WriteLine(i2);
            ///也是精度问题，数值的丢失（没有四舍五入，会直接抛弃掉后面的小数）
            
            //char和数值类型
            i2 = 'A';
            
            char c = (char)i2;
            
            //bool和string
            bool bo = true;
            ///int i4 = （int）bo；(报错）
            
            int i0 = 1;
            string str = "123";
            ///i4 =(int)str;(报错，也不可以通过括号强转）
            
            #endregion
            
            #region Parse法
            //作用：将字符串类型转换为对应的类型
            //语法：变量类型.parse("字符串")
            //注意：字符串必须能够转换成对应类型 否则报错
            
            //有符号
            string str2 = "123";
            int i4 = int.Parse("123");
            i4 = int.Parse(str2);
            Console.WriteLine(i4);
            
            ///我们填写字符串必须是要能够转成对应类型的字符，如果不符合规则，会直接卡死报错
            ///i4 = int.Parse("123.45");
            
            short s3 = short.Parse("4");
            Console.WriteLine(s3);
            
            ///填写的字符串值的范围必须是能够被变量存储的值，否则会报错
            ///short s3 = short.Parse("400000");
            
            ///把字符串的数字转变为了数字类型
            sbyte sb3 = sbyte.Parse("1");
            Console.WriteLine(sb3);
            ///简化↓意思相同
            Console.WriteLine(sbyte.Parse("1"));
            
            Console.WriteLine(long.Parse("123123"));
            
            //无符号
            Console.WriteLine(byte.Parse("1"));
            Console.WriteLine(ushort.Parse("1"));
            Console.WriteLine(ulong.Parse("1"));
            Console.WriteLine(uint.Parse("1"));
            ///Parse可以将字符串转换成变量,字符串是人类语言,变量是计算机语言
            
            //浮点数
            float f = float.Parse("1.2323");
            double d3 = double.Parse("1.2323");
            ///四舍五入，有效数字
            
            //特殊类型
            bool b5 = bool.Parse("true");
            Console.WriteLine(b5);
            
            char c2 = char.Parse("A");
            Console.WriteLine(c2);
            
            ///string里没有这个方法
            #endregion
            
            #region Convert法
            //作用：更准确地将各个类型之间进行相互转换
            //语法：Convert.To目标类型（变量或常量）
            //注意：填写地变量或常量必须正确，否则出错
            
            //转字符串 如果是把字符串转对应类型，那字符串一定要合法合规
            int a = Convert.ToInt32("12");
            ///int:4个字节，四八三十二位
            Console.WriteLine(a);
            
            //精度更准确
            a = Convert.ToInt32(1.65845f);
            Console.WriteLine(a);
            ///精度比Parse更高，因为会四舍五入，精度比强转好
            
            //特殊类型转换
            a = Convert.ToInt32(true);///1真
            Console.WriteLine(a);
            
            a= Convert.ToInt32(false);///0假
            Console.WriteLine(a);
            
            ///把bool类型转成数值类型，规则：true对应1，false对应0
            
            a = Convert.ToInt32('A');
            Console.WriteLine(a);///65
            
            ///大乱斗，各种类型都可以被转换
            
            //每一个类型都存在对应地Convert中的方法
            sbyte sb5 = Convert.ToSByte("1");
            short s5 = Convert.ToInt16("1");///short占两个字节，二八十六
            int i5 = Convert.ToInt32("1");///int占四个字节，四八三十二
            long l5 = Convert.ToInt64("1");///long占八个字节，八八六十四
            
            byte b50 = Convert.ToByte("1");
            ushort us5 = Convert.ToUInt16("1");
            uint ui5 = Convert.ToUInt32("1");
            ulong ul5 = Convert.ToUInt64("1");
            
            float f5 = Convert.ToSingle("13.2");///ToSingle为单精度的意思
            double d5 = Convert.ToDouble("13.2");
            decimal de5 = Convert.ToDecimal("13.2");
            
            bool bo5 = Convert.ToBoolean("true");
            char c5 = Convert.ToChar("A");
            
            string str5 = Convert.ToString(12345);
            
            #endregion
            
            #region 其他类型转string
            //作用：拼接打印
            //语法：变量.toString();
            
            string str6 = 1.ToString();
            str6 = true.ToString();
            str6 = 'A'.ToString();
            str6 = 1.2f.ToString();
            
            int aa = 1;
            str6 = aa.ToString();
            ///所有变量都是有对象的
            bool bo6 = true;
            str6 = bo6.ToString();
            
            Console.WriteLine("12345" + 1 + true);
            ///拼接默认将1、true自动转换为ToSrting形式，再来进行拼接
            ///当我们进行字符串拼接时，就会自动调用tostring转成string
            
            str6 = "123123" + 1 + true + 1.23;
            ///本质上是进行一个字符串的拼接
            #endregion
            
            //括号强转：低精度装高精度
            //Convert：更高的精度，各种大乱斗转
            //Parse：字符串转换为其他
            //toString：拼接转字符串
        }
    }
}
```
