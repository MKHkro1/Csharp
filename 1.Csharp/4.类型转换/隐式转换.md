# 知识点
1. 相同大类型之间的转换
2. 不同大类型之间的转换

# 讲解
```C#
//引用命名空间
using System;
//命名空间
namespace 模板
{  
    //类
    class Program
    {    //函数
         static void Main(string[] args)
        {   
            //什么是类型转换
            //不同变量类型之间的相互转换
            //隐式转换的基本规则：不同类型之间的自动转换
            
            //大核心：大范围装小范围类型
            
            #region 相同大类型之间的转化
            //有符号：sbyte、short、int、long
            //无符号：byte、ushort、uint、ulong
            //浮点数：float、double、decimal
            //特殊：bool、char、string
            
            sbyte sb = 1;
            short s = 1;
            int i = 1;
            long l = 1;
            l = i;//int隐式转换成了long
            // i = l;
            //报错（不能用大范围的装进小范围的）
            l = sb;
            l = s;
            
            byte b = 1;
            ushort us = 1;
            uint ui = 1;
            ulong ul = 1;
            ul = b;
            ul = us;
            ul = ui;
            
            //decimal是十进制，和二进制不是一个类型
            //因此decimal没有办法用隐式转化的形式去存储double和float
            decimal de = 1.1m;
            double d = 1.1;
            float f = 1.1f;
            //float可以隐式转化为double
            d = f;
            
            //特殊无大类型隐式转化（无规则）
            bool bo = true;
            char c = 'A';
            string str ="123123";
            
            #endregion
            
            #region不同类型之间的转化
            //1.无符号和有符号之间
            //有符号：
            sbyte sb2 = 1;
            short s2 = 1;
            int i2 = 1;
            long l2 = 1;
            //无符号：
            byte b2 = 1;
            ushort us2 = 1;
            uint ui2 = 1;
            ulong ul2 = 1;
            
            
            //有符号的变量不可以隐式转换成无符号的变量
            b2 = sb2;//（报错）
            //无符号的变量可以隐式转化为大范围有符号的变量
            i2 = ui2;//（报错，因为无符号的数可能会超过有符号的范围）
            i2 = b2;
            //浮点数和整数之间
            float f2 = 1.1f;
            double de = 1.1;
            decimal de2 = 1.1m;
            //浮点数可以转载任何类型的整数
            //整形转为浮点数，是存在隐式转换的
            f2 = l2;
            f2 = ul2;
            
            f2 = 10000000000000000000;//十多位的有效数字了
            Console.WriteLine(f2);//打出1E+19，科学计数法：1乘以10的19次方
            
            //虽然decimal不可以隐式存储float和double，但是可以存储整形
            //float会四舍五入造成数据丢失，所以没办法隐式转换成decimal
            //因此默认为double
            de = l2;
            de = ul2;
            
            //整数无法隐式存储浮点数，因为整数不可以存小数
            
            //其他特殊类型
            //bool什么也不能存，也不能隐式转换为其他类型
            //bool无法和其他类型相互转换
            bool bo2 =true;
            
            //char没有办法隐式存储其他类型，char可以隐式转换为整形和浮点型
            //char隐式转换成数值类型是
            //对应的数字 其实是一个ASCII码（早期的编码格式，ASCII码对应表）
            //计算机里面存储2进制
            //字符，中文，英文，标点符号 在计算机中都是一个数字
            //一个字符对应一个数字 ASCII码是一种对应关系
            char c2 ='A';
            i2 = c2;
            f2 = c2;
            ui2 = c2;
            
            //string 类型无法和其他类型进行隐式转化
            string str2 = "123";
            
            #endregion
            
            //总结：
            //高精度（大范围）装低精度（小范围）
            //double➡float➡整数（无符号、有符号）➡char
            //decimal➡整数（无符号、有符号）➡char
            //string和bool不参与隐式转换规则
        }
    }
    
}

```

# 实操
1.什么情况下会出现数据类型的隐式转换，请举例说明
```C#
//引用命名空间
using System;
//命名空间
namespace 模板
{  
    //类
    class Program
    {    //函数
         static void Main(string[] args)
        {   
            //大范围存小范围
            //变量之间的转换处理
            
            //double、float、整数、char
            //特殊：decimal、整形（无符号，有符号）、char（decimal不能存储double、float，数据丢失）
            //long、int、short、sbyte
            //ulong、uint、ushort、byte
            //无符号无法隐式存储有符号，有符号可以隐式存储小范围的无符号
            int i = 1;
            short s = 1;
            i = s;
        }
    }
    
}
```
