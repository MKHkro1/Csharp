# 知识点
1. 相同大类型之间的转换
2. 不同大类型之间的转换

# 讲解
```C#
//引用命名空间
using System;
//命名空间
namespace 模板
{  
    //类
    class Program
    {    //函数
         static void Main(string[] args)
        {   
            //什么是类型转换
            //不同变量类型之间的相互转换
            //隐式转换的基本规则：不同类型之间的自动转换
            //大范围装小范围类型
            
            #region 相同大类型之间的转化
            //有符号：sbyte、short、int、long
            //无符号：byte、ushort、uint、ulong
            //浮点数：float、double、decimal
            //特殊：bool、char、string
            
            sbyte sb = 1;
            short s = 1;
            int i = 1;
            long l = 1;
            l = i;//int隐式转换成了long
            // i = l;
            //报错（不能用大范围的装进小范围的）
            l = sb;
            l = s;
            
            byte b = 1;
            ushort us = 1;
            uint ui = 1;
            ulong ul = 1;
            ul = b;
            ul = us;
            ul = ui;
            
            //decimal是十进制，和二进制不是一个类型
            //因此decimal没有办法用隐式转化的形式去存储double和float
            decimal de = 1.1m;
            double d = 1.1;
            float f = 1.1f;
            //float可以隐式转化为double
            d = f;
            
            //特殊无大类型隐式转化（无规则）
            bool bo = true;
            char c = 'A';
            string str ="123123";
            
            #endregion
            
            #region不同类型之间的转化
            //1.无符号和有符号之间
            //有符号：
            sbyte sb2 = 1;
            short s2 = 1;
            int i2 = 1;
            long l2 = 1;
            //无符号：
            byte b2 = 1;
            ushort us2 = 1;
            uint ui2 = 1;
            ulong ul2 = 1;
            
            
            //有符号的变量不可以隐式转换成无符号的变量
            b2 = sb2;//（报错）
            //无符号的变量可以隐式转化为大范围有符号的变量
            i2 = ui2;//（报错，因为无符号的数可能会超过有符号的范围）
            i2 = b2;
            
            
            #endregion
        }
    }
    
}

```
