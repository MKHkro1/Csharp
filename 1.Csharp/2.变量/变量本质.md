## 变量的存储空间（内存）
```C#
            #region 变量的存储空间（内存）
            //1byte=8bit（一个字节，八个位）
            //1KB=1024byte
            //1MB=1024KB
            //1gb=1024mb
            //1tb=1024gb
            
            //通过sizeof方法，可以获取变量类型所占的内存空间（单位：字节）
            
            //有符号
			//通过变量赋值来获得所占的内存空间
			int sbyteSize = sizeof(sbyte); //1
			Console.WriteLine("sbyte所占的字节数为"+sbyteSize);
			int intSize = sizeof(int);//4
			Console.WriteLine("int所占的字节数为"+intSize);
			int shortSize = sizeof(short);//2
			Console.WriteLine("short所占的字节数为"+shortSize);
			int longSize = sizeof(long);//8
			Console.WriteLine("long所占的字节数为"+longSize);
			
			//无符号
			int byteSize = sizeof(byte); //1
			Console.WriteLine("byte所占的字节数为"+byteSize);
			int uintSize = sizeof(uint);//4
			Console.WriteLine("uint所占的字节数为"+uintSize);
			int ushortSize = sizeof(ushort);//2
			Console.WriteLine("ushort所占的字节数为"+ushortSize);
			int ulongSize = sizeof(ulong);//8
			Console.WriteLine("ulong所占的字节数为"+ulongSize);
			
			//浮点数
			int floatSize = sizeof(float); //4
			Console.WriteLine("float所占的字节数为"+floatSize);
			int doubleSize = sizeof(double);//8
			Console.WriteLine("double所占的字节数为"+doubleSize);
			int decimalSize = sizeof(decimal);//16
			Console.WriteLine("decimal所占的字节数为"+decmalSize);
			//内存占用的多少等于其所取值范围的大小
			
			//特殊的类型
			int boolSize = sizeof(bool); //1
			Console.WriteLine("bool所占的字节数为"+boolSize);
			int charSize = sizeof(char);//2
			Console.WriteLine("char所占的字节数为"+charSize);
			
			//sizeof不可以得到string类型所占的内存大小，因为字符串长度是可变不定的
			int stringSize = sizeof(string);//报错，变长数组（变化）
			Console.WriteLine("string所占的字节数为"+stringSize);
			
			#endregion
```
## 变量的本质
```C#
			#region 变量的本质
			//本质：二进制 一堆的0和1
			//电信号数据传递，只有开关两种状态，所以只能用0和1来表示这两种状态
			//bit：只能表示0和1两个数字，要不是0就是1
			//8bit 1byt 1byte=0000 0000 通过不同0和1的排列组合
			
			
			#endregion
			
			#region 二进制与十进制
			/*
			2 进制和 10 进制的对比
			  1  `0`
			  2  `1`
			  3  `10`
			  4  `11`
			  5  `100`
			  6
			  7
			  8
			  9
			  10
			
			
			2. 十进制转二进制规则：
            取余法转换（辗转相除法）：不停地对二取余
            从下往上看(详见下方笔记)
            
            3. 二进制转十进制的规则：
			从右往左看，最右边的一位为第 0 位，如果该位不为 0 则加上 2 的 n 位次方
			
			1：1（2<sup>0</sup>）
			2:10(0+2<sup>1</sup>)
			3:11(2<sup>0</sup>+2<sup>1</sup>)
			4:100 (0+0+2<sup>2</sup>)
			5：101（2<sup>0</sup>+0+2<sup>2</sup>）
			6：110（0+2<sup>1</sup>+2<sup>2</sup>）
			.....
			90：1011010（0+2<sup>1</sup>+0+2<sup>3</sup>+2<sup>4</sup>+0+2<sup>6</sup>）
			
			存储量
			Byte 一个字节（0000 0000）
			1111 1111：1+2+4+8+16+32+64+128=255（因此可以存储 255 个数）
			
			Int 四个字节（4 `*` 8 个零）
			结果：42 亿多
			
			有无符号
			Sbyte/byte（都是由 0000 0000 表示）
			
			Sbyte（第一个零表示为了符号位）*/
        }
    }
    
}
```
