# 讲解

```C#
//引用命名空间
using System;
//命名空间
namespace 变量
{  
    //类
    class Program
    {    //函数
         static void Main(string[] args)
        {   
            //函数语句块
            Console.WriteLine("变量");
            
            //折叠代码：编程时逻辑更为清晰
            //使用tab自动补全MyRegion,不要按space，当然，可以手动输入（VScode）
            //由#region、#endregion配对出现
            //作用：将中间包裹的代码折叠起来，避免凌乱
            //点击右侧-号可以进行折叠（VScode）
            //本质是预处理指令
            //只会在编辑时有用，执行代码时自动忽略删除
            #region MyRegion
            
            #endregion
            
            
            
            #region 如何声明变量
            //变量：可以变化的容器 存储不同类型数值的容器
            //不同变量类型 可以存储不同类型的值
            
            //变量声明固定写法：
            //变量类型 变量名 = 初始值;
            int i = 1;
            //尽量不要使用中文，发布不同软件的规则可能不一样
            
            //变量类型（14种）
            //变量名 变量命名规范
            //初始值和变量类型要统一
            //=和;是固定不变的
            //变量不要重名
            
            //变量类型（死记各种变量类型**关键字**、**存储范围**与**存储类型**）
            
            //1.有符号的整形变量（能存储 一定范围 正负数包括0的变量类型）
            //sbyte -128~127
            sbyte sb = 1;
            
            //int -21亿~21亿多
            int i2 = 21000;
            
            //short -32768~32767
            short s =3;
            
            //long -9百万兆~9百万兆
            long l =4;
            
            //潜在 可以通过+来进行拼接打印
            Console.WriteLine(sb);
            Console.WriteLine("sbyte变量sb中存储的值是:"+sb);
            
            
            //2.无符号的整形变量（可以存储一定范围0和正数的变量类型）
            //byet 0~255
            byte b = 0;
            
            //uint 0~42亿多
            uint ui =2;
            
            //ushort 0~65535之间
            ushort us = 3;
            
            //ulong 0~18百万兆
            ulong ul = 5;
            
            
            //3.浮点数（小数）
             //有效数字从左到右从非0数开始算
             
            //float 存储7/8位*有效数字*（编译器不同影响）需要加f
            //没有输入小数的时候，那么就不用加F
            float f = 0.1234567890f;
            Console.WriteLIne(f);//0.12345679(存在四舍五入)
            
            //double 存储15~17位有效数字
            //**在C#中声明一个小数默认为double类型**
            double d = 0.234567234890;
            Console.WriteLIne(d);
            
            //decimal 存储27~28位的有效数字，不建议使用,需要加m
            decimal de = 0.1245467866356434m;
            Console.WriteLIne(de);
            
            //4.特殊类型
            //bool ture false(表示真假) 真假类型
            bool bo = true;
            bool bo2 = false;
            //如果需要自动转成字符串进行拼接则需要_进行拼接，在此处可以让true，false转换为文字
             Console.WriteLIne(bo+"_"+bo2);
             
            //char （存储单个字符,需要接'单引号） 字符类型
            char c = '唐';
            Console.WriteLIne(c);
            
            //string（字符串类型，用来存储无上限的字符,需要接""双引号）
            string str = "你好，世界！";
            
            
            //变量不可以无中生有，必须要声明才能用
            int X =100 ;
            //变量的使用
            Console.WriteLine(X);
            //变量的修改
            X = 29;
            #endregion
            
            #region 为什么有这么多不同的变量类型
            //不同的变量，存储的范围和类型不一样，本质是占用的内存空间不同
            //选择不同的数据（变量）类型装在不同的数据
            
            //eg:
            //姓名
            string myName ="欧阳萱畅";
            
            //年龄
            byte age = 21;
            
            //身高
            float height = 161.0f;
            
            //体重
            float weight =52.0f;
            
            //性别 (自己定义：true 男，flase 女）
            bool sex = flase;
            
            //简便：数字用int，小数用float，字符串用string，真假用bool
            
            #endregion
            
            #region 多个相同类型变量如何同时声明
            int i3 = 1;
            float f2 = 3.2f;
            string str2 ="这是一段文字";
            bool bo2 = true;
            
            //使用逗号隔开,同时声明
            //固定写法:
            //变量类型 变量名 = 初始值，变量名 = 初始值，....;
            int a1 = 1 ,b1 = 2,c1 = 3,d1 = 4;
            string s1 ="我是一段文字",s2 = "我还是一段文字";
            //字符串可以拼接打印
            Console.WriteLine(s1 + s2);
            
            #endregion
            
            
            
            
            #region 变量初始化相关
            int a2;//此变量无初始值，此变量不安全，使用之前必须要赋值，不建议如此使用
            a2 = 5
            Console.WriteLine(a2);
            
            #endregion
            
        }
    }
    
}

```
