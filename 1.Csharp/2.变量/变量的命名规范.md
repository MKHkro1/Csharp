# 讲解
```C#
//引用命名空间
using System;
//命名空间
namespace 变量的命名规范
{  
    //类
    class Program
    {    //函数
         static void Main(string[] args)
        {       
            #region 必须遵守的规则
            /*
            1.不能重名
            2.不能以数字开头
            3.不能使用程序关键字命名
            4.不能有特殊符号（下划线除外）
            */
            int i = 1;//(成功)
            string i = "123";//(报错)
            int 2i = 2;//(报错)
            int float = 2;//(报错)
            int a? = 3;//(报错)
            int a_2 = 5;//(成功)
            
            //建议的命名规则：变量名要有含义➡用英文（拼音）表示变量的作用
            //非常不建议的命名规则：用汉字命名（在程序直接的运作也许会出现不兼容的情况）
            
            #endregion
            
            #region 常用的命名规则
            //驼峰命名法：首字母小写，之后单词首字母大写（变量）
            string myName = "欧阳萱畅";
            string yourName = "你好";
            string yourMotherName = "你好";
            int yourMotherage = 30;
            
            //帕斯卡命名法——所有单词首字母都大写（函数、类）
            string MyName = "阿巴阿巴";
            
            //潜在知识点：C#对大小写是敏感的，是区分的
            
            #endregion
            
            
        }
    }
    
}

```
