## 位与 &

```C#
//规则 连接两个数值进行位计算 将数值转为2进制
//对位运算 有0则0 同1为1
int a = 1;//001
int b = 5;//101
int c = a & b;//需要使用起来

//8421法，记下128 64 32 16 8 4 2 1八位数，19=16+2+1，所以在相应位数填1，即10011

//&位与运算
//001
//101
//001
//结果是1
Console.WriteLine(c);

a = 3;//011
b = 19;//10011
c = a & b;
//  011(没有则补0)
//10011
//00011
//结果是3
Console.WriteLine(c);

//多个数值进行位与运算时 没有括号的情况下 从左到右依次计算

int d = 1 & 5 & 19;
//  001
//  101
//  001

//  001
//10011
//00001
//结果是1

a = 1;//001
b = 2;//010
Console.WriteLine(a & b);//000 结果是0
```
