# 字符串拼接方法 1
```C#
#region 字符串拼接方式1
//之前的算数运算符 知识用来数值类型变量进行数学运算的
//而 string 不存在算数运算符不能计算 但是可以通过+号来进行字符串的拼接
string str = "123";
//其实+号进行字符串拼接
str = str + "456";
Console.WriteLine(str);

str = str + 1; //1会被自动转换成字符串

// 复合运算符 +=
str = "123";
str += "1" + 4 + true;
Console.WriteLine(str);
//都是会被转换成字符串进行拼接

str += 1 + 2 + 3 + 4;// 12310
// 1+2+3+4 会被当成数学运算进行计算 计算完成后再向左边赋值
//1234都是int型，只有10在给str时才发现类型不对进行了类型转换

str += "" + 1 + 2 + 3 + 4;// 1231234
// ""是字符串类型，后面的1+2+3+4都会被当成字符串进行拼接

Console.WriteLine(str);

str = "";
str += 1 + 2 + "" + 3 + 4;
Console.WriteLine(str);
str += 1 + 2 + "" + ( 3 + 4 );
//看字符串和数字的前后顺序，只要碰见字符串则后面的数字都变成字符串，碰不见那就算了
//如果用单引号开始也会被转成数字加入运算而不是字符拼接，而且true既不是1，也不是字符串不能和运算时的＋放一起

int a = '1' + 2 + 3 + 4;// 58
//在C++中，字符常量如'1'实际上是一个整数，代表该字符的ASCII值。ASCII值中，字符'1'的十进制值是49。
Console.WriteLine(a);

str = "123";
str = str + 1 + 2 + 3;
Console.WriteLine(str);//123123
//遵循等号右边从左到右的顺序进行计算，然后把结果赋值给左边（运算符思维）

str = str + ( 1 + 2 + 3 );
//注意：用+号拼接 是用符号的唯一方法 不能用-*/%...
#endregion
```

# 字符串凭借方法 2
```C#
#region 字符串拼接方式2
//固定语法
//string.Format("待拼接的内容",内容1,内容2,...);
//凭借内容中的固定规则
//想要被拼接的内容用占位符替代{数字} 数字：0~n 依次往后

string str2 = string.Format("我是{0}，今年{1}岁了, 我想要{2}", "萱", 21, "好好学习");
//生成一个拼接的内容

Console.WriteLine(str2);

str2 = string.Format("{0}{1}{2}", 1,true,false);//默认都调用了ToString()方法
Console.WriteLine(str2);

#endregion

```

# 控制台打印拼接
```C#
#region 控制台打印拼接
Console.WriteLine("asdf{0},{1},sddfasdf{2}",1,true,false);
//与方法2一模一样规范，是可以直接可以使用
//Console.WriteLine()方法本身就有重载版本支持这种用法
//WriteLine专门提供了这种方法

Console.Write("asdf{0},{1},sddfasdf{2}", 1, true, false);
//多填不会报错，少填会报错，没有对应的项
//占位符如果没有从 {0} 开始写也会报错
#endregion
```
# 练习题
1. 定义一个变量存储客户的姓名，然后在屏幕上显示：“你好，XXX”。
2. 定义两个变量，一个存储客户的姓名，另一个存储年龄，然后再屏幕上显示: "xxx+yyy岁了”。xxx代表客户的姓名，yyy代表年龄举例(唐老狮18岁了) 
3. 当我们去面试时，前台会要求我们填一张表格，有姓名，年龄，邮箱， 家庭住址，期望工资，请把这些信息在控制台输出。 
4. 请用户输入用户名、年龄、班级，最后一起用占位符形式打印出来
```C#
#region 练习题
//1
Console.WriteLine("你好，{0}", "张三");

//2
try
{
    Console.WriteLine("请输入你的名字：");
    string userName = Console.ReadLine();
    Console.WriteLine("请输入你的年龄：");
    int userAge = int.Parse(Console.ReadLine());
    Console.WriteLine("{0}，你今年{1}岁了", userName, userAge);
}
catch (Exception e)
{
    Console.WriteLine("出现异常：" + e.Message);
}
finally
{
    Console.WriteLine("程序执行完毕");
}

//3
Console.WriteLine("{0}{1}{2}{3}{4}", "姓名","年龄","邮箱","家庭住址","期望工资");

//4
try
{
    Console.WriteLine("请输入你的名字：");
    string userName = Console.ReadLine();
    Console.WriteLine("请输入你的年龄：");
    int userAge = int.Parse(Console.ReadLine());
    Console.WriteLine("请输入你的班级：");
    int userClass = int.Parse(Console.ReadLine());
    Console.WriteLine("{0}，你今年{1}岁了，你在{2}班", userName, userAge,userClass);
}
catch (Exception e)
{
    Console.WriteLine("出现异常：" + e.Message);
}
finally
{
    Console.WriteLine("程序执行完毕");
}

#endregion
```
# 答案
```C#
#region 答案
//1
string name = "萱";
Console.WriteLine("你好，{0}", name);

string str0 = string.Format("你好，{0}", name);
Console.WriteLine(str0);

//2
string name2 = "萱";
int age2 = 21;

Console.WriteLine(name2 + age2 + "了");

str0 = string.Format("{0},{1}了", name2, age2);

Console.WriteLine("{0},{1}了", name2, age2);

int age3 = 21;

    name3, age3, email, address, money);
```
