## 转移字符的使用
```C#
            #region 转义字符的使用
            /*
            什么是转义字符？
            用于表示特殊含义的字符，是字符串的一种
            eg：在字符中表示出 引号 单引号 空行等   
            
            固定写法：
            \加字符用来表示不同含义的字符
            */
            
            //*单引号(\')
            string str = "\' 哈哈哈\'";
            Console.WriteLine(str);
            
            //*双引号(\")
            str = "\"哈哈哈\"";
            Console.WriteLine(str);
            
            //*换行（\n）
            //直接按enter的话
            str = "123" +
            "123";
            //还是会直接拼接补全
            Console.WriteLine(str);
            //正确用法:
            str = "1234\n1234";
            Console.WriteLine(str);
            
            //*斜杠（\\）
            str = "1234\\1234";
            Console.WriteLine(str);
            //斜杆可以使用char变量单独输出
            //计算机的文件路径是需要用到\
            
            //*制表符（\t）
            str = "1234\t1234";
            Console.WriteLine(str);
            //空一个Tab键盘，快速地空四格，多用于程序的对齐
            
            //*光标退格（\b）
            str = "1234\b1234";
            Console.WriteLine(str);
            //输出一个\b就可以删除前面字符
            
            //*空字符（\0）
            str = "1234\0123";
            Console.WriteLine(str);
            //占位用，别的语言常用
            
            //*警报音（\a）
            str = "\a";
            Console.WriteLine(str);
            
            //在打印中直接输出字符串的时候也可以带转义字符
            Console.WriteLine("123\0123\t1235\\123\a");
            #endregion
            
            #region 取消转义字符
            string str2 = @"哈哈哈\哈哈哈";
            Console.WriteLine(str2);
            //@取消后面字符串的转移字符
            
            Console.WriteLine(@"\n\\");
            //输出的是 \n\\
            #endregion
        }
    }
    
}

```
