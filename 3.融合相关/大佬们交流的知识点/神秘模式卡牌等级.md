```C#
 public static CardLevel GetCardLevel(PlantType thePlantType)
{
    // 检查是否为红色植物（最高等级）
    if (TypeMgr.RedPlant.Contains(thePlantType))
    {
        return CardLevel.Level5; // 等级5
    }
    
    // 检查是否为基础植物（最低等级）
    if (Lawnf.IsBasicPlant(thePlantType))
    {
        return CardLevel.Level0; // 等级0
    }
    
    // 检查是否为终极植物
    if (Lawnf.IsUltiPlant(thePlantType))
    {
        // 如果是金色终极植物，返回等级4，否则返回等级3
        return TravelMgr.GoldUltiPlants.Contains(thePlantType) 
            ? CardLevel.Level4  // 金色终极植物
            : CardLevel.Level3; // 普通终极植物
    }
    
    // 检查是否为超级植物
    if (Lawnf.IsSuperPlant(thePlantType))
    {
        return CardLevel.Level2; // 等级2
    }
    
    // 默认情况
    return CardLevel.Level1; // 等级1
}
```
