# 讲解
```C#
//引用命名空间
using System;
//命名空间
namespace 变量的命名规范
{  
    //类
    class Program
    {    //函数
         static void Main(string[] args)
        {       
            #region 必须遵守的规则
            /*
            1.不能重名
            2.不能以数字开头
            3.不能使用程序关键字命名
            4.不能有特俗符号（下划线除外）
            */
            int i = 1;//(成功)
            string i = "123";//(报错)
            int 2i = 2;//(报错)
            int float = 2;//(报错)
            int a? = 3;//(报错)
            int a_2 = 5;//(成功)
            
            
            #endregion
            
            
        }
    }
    
}

```
