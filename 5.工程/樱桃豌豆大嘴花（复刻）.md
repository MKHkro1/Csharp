namespace CherryPeaChomper.BepInEx  
{  
    public class CherryPeaChomper : MonoBehaviour  
    {  
        public CherryPeaChomper() : base(ClassInjector.DerivedConstructorPointer<CherryPeaChomper>())  
        {            
        	ClassInjector.DerivedConstructorBody(this);  
        }  
        public CherryPeaChomper(IntPtr i) : base(i) { }  
        
        private float lastCherryTime = 0f;  
        private float cherryInterval = 0.5f;  
        
        public PeaChomper plant => base.gameObject.GetComponent<PeaChomper>();  
        
        public void Awake()  
        {            
        	this.plant.shoot = base.transform.Find("Shoot");  
            lastCherryTime = Time.time;  
        }  
        void Update()  
        {            
        	if (Time.time - lastCherryTime >= cherryInterval)  
            {                
            	GenerateCherryBomb();  
                lastCherryTime = Time.time;  
            }        
        }  
        
        public void CustomAnimShoot()  
        {            
        	if (this.plant?.shoot != null)  
            {                
            	CreateCherryBomb(this.plant.shoot.transform.position + Vector3.right * 0.5f);  
            }        
        }  
        
        public Bullet AnimShooting(Vector3 position)  
        {            CreateCherryBomb(position + Vector3.right * 0.5f);  
            return null;  
        }  
        
        public void Bite()  
        {            
        	if (this.plant?.shoot != null)  
            {                
            	CreateCherryBomb(this.plant.shoot.transform.position);  
                ScreenShake.shakeDuration = 0.15f;  
                GameAPP.PlaySound(40, 0.5f, 1f);  
            }        
        }  
        
        private void GenerateCherryBomb()  
        {            
        	if (this.plant?.shoot != null)  
            {                
            	Vector3 pos = this.plant.shoot.transform.position + Vector3.right * 1.0f;  
                CreateCherryBomb(pos);  
                GameAPP.PlaySound(40, 0.25f, 0.8f);  
            }        
        }  
        
        private void CreateCherryBomb(Vector3 position)  
        {            
        	if (GameAPP.particlePrefab?.Length > 2 && GameAPP.particlePrefab[2] != null)  
            {                
            	GameObject bomb = Object.Instantiate(GameAPP.particlePrefab[2], position, Quaternion.identity);  
                BombCherry bombComponent = bomb?.GetComponent<BombCherry>();  
                
                if (bombComponent != null)  
                {                    
                	bombComponent.bombRow = this.plant.thePlantRow;  
                }            
            }        
        }    
    }
}
