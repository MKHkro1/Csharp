# 讲解
```C#
//引用命名空间
using System;
//命名空间
namespace 模板
{  
    //类
    class Program
    {    //函数
         static void Main(string[] args)
        {   
            #region 常量的申明
            Console.WriteLine("常量");
            //关键词：const
            //固定写法：
            //const 变量类型 变量名 = 初始值;
            
            int i = 10;
            
            const int i2 = 20;//i2是常量
            
            #endregion
            
            #region 常量的特点
            //1.必须初始化
            //2.不能被修改
            
            string name;
            name = "123";
            
            const string myName ="欧阳萱畅";
            
            const string myName2;//错误，必须要初始化
            
            myName = “123”;//这个是常量，无法被修改
            
            
            //作用：申明一些常用不变的变量
            
            //PI ：3.1415926...
            const float PI = 3.1415926f;//公理性质
            
            Console.WriteLine(PI);//避免被修改
            
            #endregion
        }
    }
    
}

```
